// Ein paar Punkte definieren
A(100|150), B(240|200), C(190|90).

// Mitte zwische zwei Punkten bestimmen
M = proc(A,B)
    A°B, B°A,
    {X,Y} <- A°B n B°A; 
    {P} <- X--Y n A--B;
end;

// Orthogonale Projektion von P auf A--B
Proj = proc(A,B,P)
	K = (M(A,P))°A,
	{X} <- (K n A--B) \ {A};
end;

A--B,
"X"=Proj(A,B,C),
A-C,C-X,X-A.
